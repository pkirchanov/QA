
pwd 		// посмотреть где я нахожусь

cd 		// изменить директорию

cd ..		// перейти в предыдущую директорию
cd ../../.. 	// перейти на несколько директорий вверх

ls		// посмотреть содержимое папки
ls -l		// содержимое со деталями по файлам - размер и тд
ls -a		// 
ls -la 		// посмотреть содержимое папки (со скрытыми файлами?)

vim		// открыть тектовый редактор
vim file.txt	// открыть файл file.txt

git push                  	// пушим изменения на внешний дипозторий

git pull - слить изменеия с github на локальный депозиторий
